import pandas as pd #Импорт библиотеки Pandas
import numpy as np #Импорт библиотеки Numpy
import matplotlib.pyplot as plt #Импорт библиотеки Matplotlib
#from sklearn.linear_model import LinearRegression, GammaRegressor
import csv

pd.set_option('display.max_columns', None) #Убираем ограничение с колонок, при выводе БД
rd = pd.read_csv('test_ses.csv', delimiter=',') #Открываем БД, и ставим разделитель, что бы можно было вывести БД
db = pd.DataFrame(rd) #Переменная, при обращении к которой, будет выводится БД

print(db.describe()) #Вывод информации по БД, для того что бы найти пустые значения если они есть.
db.fillna(0) #Заменяем пустые значения на 0
db.sort_values('site1') #Сортировка таблицы по колонке 'site1'
print(db.describe()) #Перепроверяем, убрались ли пустые значения

#Выводим график БД
plt.scatter(db['site1'], db['site2']) #Указываем тип графика, и выбираем колонки из БД
plt.show() #Выводим точечный график
